[docs]    def data_to_eggroll_table(self, data, jobid, partition=10, work_mode=0):
        eggroll.init(jobid, mode=work_mode)
        data_table = eggroll.parallelize(data, include_key=False, partition=partition)
        return data_table



[docs]    def table_to_list(self, table_instance):
        res_list = []

        return res_list
